import random
last_price = 4430
num_simulations = 100000
num_days = 252

simulation_df = pd.DataFrame()

finalprices = []

for x in range(num_simulations):
    count = 0
    
    price_series = []
    
    price = last_price*(1+random.choice(SP_3.Close))
    price_series.append(price)
    
    for y in range(num_days):
        if count == 251:
            break
        price = price_series[count]*(1+random.choice(SP_3.Close))
        price_series.append(price)
        
        count += 1   
    simulation_df[x] = price_series
    
    finalprices.append(price_series[251])
    #print(finalprices)

fig = plt.figure()
plt.plot(simulation_df)
plt.show()
